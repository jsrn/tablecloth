<div>
  <h3>Ingredients</h3>

  <ul>
    <li>25ml gin</li>
    <li>Tonic water</li>
    <li>Cooled black tea</li>
    <li>Twist of Lemon rind</li>
  </ul>
</div>

<div>
  <h3>Method</h3>

  <ul>
    <li>Add gin to highball glass</li>
    <li>Add tonic water and tea in 2:1 ratio</li>
    <li>Add rind to garnish</li>
  </ul>
</div>
