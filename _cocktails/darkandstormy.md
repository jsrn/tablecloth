<div>
	<h3>Ingredients</h3>

	<ul>
		<li>2oz Dark Rum</li>
		<li>3oz Ginger Beer</li>
		<li>1 citrus wedge</li>
		<li>Ice</li>
	</ul>
</div>

<div>
	<h3>Method</h3>

	<ul>
		<li>Fill glass with ice</li>
		<li>Mix rum and ginger beer in glass</li>
		<li>Garnish with citrus wedge</li>
	</ul>
</div>
